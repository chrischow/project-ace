function ProgressCard(props) {
    // Define styles
    var styleCircleText = 'progress-card--circle-text';
    var styleHeaderText = 'progress-card--header-text';
    var styleMetric = 'progress-card--metric';
    var styleMetricTitle = 'progress-card--metric-title';
    var styleMetricBetween = 'progress-card--metric-between';

    // Add "-sm" suffix for Team cards
    if (props.isTeam) {
        styleCircleText = styleCircleText + '-sm';
        styleHeaderText = styleHeaderText + '-sm';
        styleMetric = styleMetric + '-sm';
        styleMetricTitle = styleMetricTitle + '-sm';
        styleMetricBetween = styleMetricBetween + '-sm';
    }

    return (
        <div>
            <div className="row justify-content-center align-items-center">
                <div className={props.isTeam ? "col-5" : "col-6"}>
                    <div id={props.progressId} className="progress-circle text-center"></div>
                    <div className={styleCircleText + " text-center mt-3"}>
                        AVG Objective Progress
                    </div>
                </div>
                <div className={props.isTeam ? "col-7" : "col-6"}>
                    <h4 className={styleHeaderText + " mb-3"}>Objectives</h4>
                    <div className="row mb-4 align-items-center justify-content-center">
                        <div className="col-5 text-center">
                            <div className={styleMetric}>{props.data.objective.inProgress}</div>
                            <div className={styleMetricTitle}>In Progress</div>
                        </div>
                        <div className="col-1 text-center">
                            <div className={styleMetricBetween}>of</div>
                        </div>
                        <div className="col-5 text-center">
                            <div className={styleMetric}>{props.data.objective.total}</div>
                            <div className={styleMetricTitle}>Total</div>
                        </div>
                    </div>
                    <hr />
                    <h4 className={styleHeaderText + " mt-4 mb-3"}>Key Results</h4>
                    <div className="row mb-3 align-items-center justify-content-center">
                        <div className="col-5 text-center">
                            <div className={styleMetric}>{props.data.keyResults.inProgress}</div>
                            <div className={styleMetricTitle}>In Progress</div>
                        </div>
                        <div className="col-1 text-center">
                            <div className={styleMetricBetween}>of</div>
                        </div>
                        <div className="col-5 text-center">
                            <div className={styleMetric}>{props.data.keyResults.total}</div>
                            <div className={styleMetricTitle}>Total</div>
                        </div>
                    </div>
                </div>
            </div>
        </div>
    )
}
