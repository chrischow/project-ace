function ObjectiveCard(props) {
    const progress = Number(props.progress);
    const progressNow = Math.round(100 * progress);

    return (
        <div className="objective-card">
            <h4 className="objective-card--tag">Objective</h4>
            <h3 className="objective-card--title">
                {props.title}
            </h3>
            <p className="objective-card--description">
                {props.description}
            </p>
            <div className="progress">
                <div
                    className="progress-bar"
                    role="progressbar"
                    aria-valuemin="0"
                    aria-valuemax="100"
                    aria-valuenow={progressNow}
                    style={{width: progressNow+'%'}}
                ></div>
            </div>
        </div>
    )
}
